# -----------------------------------------------------------------------------
# HARDWARE COMMUNICATION
# -----------------------------------------------------------------------------

pymodbus>=3.5.0
# Used for: ModbusSerialClient (RTU mode) — communicates with:
#             • Relay controller  (slave ID 2, registers 0–3)
#             • Galvo Motor X/Y   (slave ID 2, registers 0–6, float32)
#             • Z-axis controller (slave ID 2)
#             • XYZ stereotaxic stage (slave ID 1, register address 1)
#           Protocol: 9600 baud, 8N1, timeout=10s (galvo/relay/z-axis).
#           XYZ stage also 9600 baud.

pyserial>=3.5
# Used for: serial.tools.list_ports (COM port auto-detection),
#           serial.Serial for laser current monitor (19200 baud, ASCII,
#           reads lines like "Current: 450.00").
